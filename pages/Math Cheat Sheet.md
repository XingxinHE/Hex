	- $(1+x)^r$ near $x=0$  
	  $$
	  \begin{align}
	  1+rx+\frac{r(r-1)}{2}x^2+O(x^3)
	  \end{align}
	  $$
	- $\sin{(x)}$ near $x=0$  
	  $$
	  \begin{align}
	  x+O(x^3)
	  \end{align}
	  $$
	- $\cos{(x)}$ near $x=0$  
	  $$
	  \begin{align}
	  1-\frac{x^2}{2}+O(x^3)
	  \end{align}
	  $$
	- $e^x$ near $x=0$  
	  $$
	  \begin{align}
	  1+x+\frac{x^2}{2}+O(x^3)
	  \end{align}
	  $$
	- $\ln{(1+x)}$ near $x=0$  
	  $$
	  \begin{align}
	  x-\frac{x^2}{2}+O(x^3)
